18  /**
19   * This class is the super class of all Mojos that generates the code, that is the {@link GenerateServerCodeMojo} and
20   * the {@link GraphQLMojo} mojos. It contains all parameters that are common to these goals. The parameters common to
21   * all goal are inherited from the {@link AbstractGenerateCodeCommonMojo} class.<BR/>
22   * This avoids to redeclare each common parameter in each Mojo, including its comment. When a comment is updated, only
23   * one update is necessary, instead of updating it in each
24   * 
25   * @author etienne-sf
26   */

31  	 * <P>
32  	 * This parameter is used only when generateBatchMappingDataFetchers is set to <i>true</i>. It determines the return
33  	 * type of the data fetchers, as defined in the <a href=
34  	 * "https://docs.spring.io/spring-graphql/reference/controllers.html#controllers.batch-mapping.return.values">spring-graphql
35  	 * documentation</a>.
36  	 * </P>
37  	 * <P>
38  	 * The allowed values are (where K is the key type, that is: the parent object, and V is the value to be loaded in
39  	 * batch):
40  	 * </P>
41  	 * <TABLE>
42  	 * <ROW>
43  	 * <TH>Value</TH>
44  	 * <TH>Return type</TH></ROW> <ROW>
45  	 * <TD>MONO_MAP</TD></TD>Mono&lt;Map&lt;K,V&gt;&gt;</TD></ROW> <ROW>
46  	 * <TD>MAP</TD>
47  	 * <TD>Map&lt;K,V&gt;</TD></ROW> <ROW>
48  	 * <TD>FLUX></TD>
49  	 * <TD>Flux&lt;V&gt;</TD></ROW> <ROW>
50  	 * <TD>COLLECTION</TD>
51  	 * <TD>Collection&lt;V&gt;</TD></ROW>
52  	 * </TABLE>
53  	 * <P>
54  	 * The default value is <code>Flux&lt;V&gt;</code>
55  	 * </P>
56  	 * <P>
57  	 * For an easier use of this parameter, the comment of the generated data fetchers details the exact expected type.
58  	 * </P>
59  	 */
60  	@Parameter(property = "com.graphql_java_generator.mavenplugin.batchMappingDataFetcherReturnType", defaultValue = GraphQLConfiguration.DEFAULT_BATCH_MAPPING_DATA_FETCHER_RETURN_TYPE_STR)

63  	/**
64  	 * <P>
65  	 * (only for server mode) Indicates if the plugin should generate add the {@link BatchLoaderEnvironment} parameter
66  	 * to the <I>batchLoader</I> methods, in DataFetchersDelegate. This parameter allows to get the context of the Batch
67  	 * Loader, including the context associated to the id, when using the id has been added by the
68  	 * {@link org.dataloader.DataLoader#load(Object, Object)} method.
69  	 * </P>
70  	 * <P>
71  	 * For instance, if you have the method below, for a field named <I>oneWithIdSubType</I> in a DataFetcherDelegate:
72  	 * </P>
73  	 * 
74  	 * <PRE>
75  	 * &#64;Override
76  	 * public CompletableFuture&lt;AllFieldCasesWithIdSubtype> oneWithIdSubType(
77  	 * 		DataFetchingEnvironment dataFetchingEnvironment, DataLoader&lt;UUID, AllFieldCasesWithIdSubtype> dataLoader,
78  	 * 		AllFieldCases source, Boolean uppercase) {
79  	 * 	return dataLoader.load(UUID.randomUUID());
80  	 * }
81  	 * </PRE>
82  	 * <P>
83  	 * then, in the <I>AllFieldCasesWithIdSubtype</I> DataFetcherDelegate, you can retrieve the uppercase this way:
84  	 * </P>
85  	 * 
86  	 * <PRE>
87  	 * &#64;Override
88  	 * public List&lt;AllFieldCasesWithIdSubtype> batchLoader(List&lt;UUID> keys, BatchLoaderEnvironment environment) {
89  	 * 	List&lt;AllFieldCasesWithIdSubtype> list = new ArrayList<>(keys.size());
90  	 * 	for (UUID id : keys) {
91  	 * 		// Let's manage the uppercase parameter, that was associated with this key
92  	 * 		Boolean uppercase = (Boolean) environment.getKeyContexts().get(id);
93  	 * 		if (uppercase != null && uppercase) {
94  	 * 			item.setName(item.getName().toUpperCase());
95  	 * 		}
96  	 * 
97  	 * 		// Do something with the id and the uppercase value
98  	 * 	}
99  	 * 	return list;
100 	 * }
101 	 * </PRE>
102 	 * <P>
103 	 * For more complex cases, you can store a {@link Map} with all the needed values, instead of just the parameter
104 	 * value.
105 	 * </P>
106 	 * <P>
107 	 * <B><I>The default value changed since 2.0 version: it is false in 1.x version, and true since the 2.0
108 	 * version</I></B>
109 	 * </P>
110 	 */
111 	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateBatchLoaderEnvironment", defaultValue = GraphQLConfiguration.DEFAULT_GENERATE_BATCH_LOADER_ENVIRONMENT)
112 	public boolean generateBatchLoaderEnvironment;

114 	/**
115 	 * <P>
116 	 * If this parameter is set to <i>true</i>, the spring GraphQL controller methods will be annotated with the
117 	 * <code>@BatchMapping</code> (instead of the <code>@SchemaMapping</code>). This allows to manage the of the N+1
118 	 * select problem: so this allows much better performances, by highly diminishing the number of executed requests
119 	 * (avoid to execute several times the same "sub-query")
120 	 * <P>
121 	 * </P>
122 	 * When setting this parameter to <i>true</i>, the main changes are:
123 	 * </P>
124 	 * <UL>
125 	 * <LI>The <code>@BatchMapping</code> annotation may be applied to all data fetchers without argument(s) that return
126 	 * either a List, a Type, an Interface or an Union.</LI>
127 	 * <LI>The return type must be defined in the controller: it may not be `Object`, as spring-graphql builds the
128 	 * proper BatchLoader while loading the controllers, when the server starts. The return type for this method is
129 	 * managed by the <code>batchMappingMethodReturnType</code> plugin parameter</LI>
130 	 * <LI>DataLoader is managed transparently by spring (instead of having to declare it in the generated controller,
131 	 * and having it as a parameter in the generated data fetchers)</LI>
132 	 * <LI>The batch mapping is generalized on all data fetchers</LI>
133 	 * <LI>The <code>DataFetchersDelegate</code> method's signature changes</LI>
134 	 * <LI>The <code>generateBatchLoaderEnvironment</code>, <code>generateDataFetcherForEveryFieldsWithArguments</code>
135 	 * and <code>generateDataLoaderForLists</code> plugin parameters are ignored</LI>
136 	 * </UL>
137 	 * <P>
138 	 * A typical method signature for a data fetcher would be as below, where the return type is controller by the
139 	 * <code>batchMappingMethodReturnType</code> plugin parameter :
140 	 * </P>
141 	 * 
142 	 * <PRE>
143 	 * public Flux<Topic> topics(//
144 	 * 		BatchLoaderEnvironment batchLoaderEnvironment, //
145 	 * 		GraphQLContext graphQLContext, //
146 	 * 		List<Board> boards);
147 	 * </PRE>
148 	 * <P>
149 	 * Please note that the <code>@BatchMapping</code> annotation is a shortcut to avoid boilerplate code, for the most
150 	 * common cases. See <a href="https://github.com/spring-projects/spring-graphql/issues/232">this discussion</a> for
151 	 * more information on this. For most complex cases, the use of a DataLoader is recommended by the spring-graphql
152 	 * case. And in these cases, the plugin will generate a method with the <code>@SchemaMapping</code> annotation
153 	 * </P>
154 	 */
155 	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateBatchMappingDataFetchers", defaultValue = GenerateServerCodeConfiguration.DEFAULT_GENERATE_BATCH_MAPPING_DATA_FETCHERS)
156 	public boolean generateBatchMappingDataFetchers;

159 	 * <P>
160 	 * (only for server mode, since 2.5) Defines if a data fetcher is needed for every GraphQL field that has input
161 	 * argument, and add them in the generated POJOs. This allows a better compatibility with spring-graphql, and an
162 	 * easy access to the field's parameters.
163 	 * </P>
164 	 * <P>
165 	 * This parameter is <b>forced to true</b> when the <code>generateBatchMappingDataFetchers</code> parameter is set
166 	 * to <i>true</i>.
167 	 * </P>
168 	 * <P>
169 	 * With this argument to false, the data fetchers are generated only for field which type is a type (not a scalar or
170 	 * an enum), and for the query, mutation and subscription types.
171 	 * </P>
172 	 * <P>
173 	 * With this argument to true, the data fetchers are generated for all GraphQL fields which type is a type (not a
174 	 * scalar or an enum) <b><i>or</i></b> that has one or arguments
175 	 * </P>
176 	 * <P>
177 	 * This parameter is available since version 2.5:
178 	 * </P>
179 	 * <UL>
180 	 * <LI>From 2.5 to 3.0: Default value is false in 2.x versions for backward compatibility with existing
181 	 * implementations based on the plugin. But the <b>recommended value is true</b>.</LI>
182 	 * <LI>From 3.0.1: Default value is true</LI>
183 	 * </UL>
184 	 */
185 	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateDataFetcherForEveryFieldsWithArguments", defaultValue = GenerateServerCodeConfiguration.DEFAULT_GENERATE_DATA_FETCHER_FOR_EVERY_FIELD_WITH_ARGUMENT)

203 	/**
204 	 * <P>
205 	 * (only for server mode) Defines how the methods in the data fetchers delegates are generated. The detailed
206 	 * information is available in the
207 	 * <a href="https://github.com/graphql-java-generator/graphql-maven-plugin-project/wiki/server">Wiki server page</a>
208 	 * </P>
209 	 * <P>
210 	 * When generateDataLoaderForLists is false (default mode), the data loaders are used only for fields that don't
211 	 * return a list. In other words, for fields which type is a sub-object with an id, two methods are generated: one
212 	 * which returns a {@link CompletableFuture}, and one which returns a none {@link CompletableFuture} result (that is
213 	 * used by the generated code only if no data loader is available).
214 	 * </P>
215 	 * <P>
216 	 * When generateDataLoaderForLists is true, the above behavior is extended to fields that are a list.
217 	 * </P>
218 	 * <P>
219 	 * Note: if set to true, this plugin parameter make the use of data loader mandatory for every field which type is a
220 	 * list of GraphQL objects, which have an id. This may not be suitable, for instance when your data is stored in a
221 	 * relational database, where you would need a first query to retrieve the ids and push them into the data loader,
222 	 * then another one to retrieve the associated values. If you want to use data loader for only some of particular
223 	 * fields, you should <b>consider using the <code>generateDataLoaderForLists</code></b>. You'll find more
224 	 * information on the
225 	 * <a href="https://github.com/graphql-java-generator/graphql-maven-plugin-project/wiki/server">Wiki server
226 	 * page</a>.
227 	 * </P>
228 	 * <P>
229 	 * This parameter is available since version 1.18.4
230 	 * </P>
231 	 */
232 	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateDataLoaderForLists", defaultValue = GraphQLConfiguration.DEFAULT_GENERATE_DATA_LOADER_FOR_LISTS)
233 	boolean generateDataLoaderForLists;

235 	/**
236 	 * <P>
237 	 * This parameter marks a list of GraphQL mappings as ignored, so that they are not generated by the plugin. These
238 	 * ignored mappings can then be defined by the specific implementation.
239 	 * </P>
240 	 * <P>
241 	 * The other way to it is to create a spring GraphQL Controller, that overrides the controller generated by the
242 	 * plugin. But this may lead to this error:
243 	 * <code>Ambiguous mapping. Cannot map 'xxxController' method [...] to 'Type.field': there is already 'yyy' bean method [...] mapped.</code>
244 	 * </P>
245 	 * <P>
246 	 * The parameter may contain:
247 	 * </P>
248 	 * <UL>
249 	 * <LI>The '*' character: this would mark all controllers and DataFetchersDeleagate to be ignored. That is: none
250 	 * would be generated, and it's up to the specific implementation to manage them. In this mode, you still benefit of
251 	 * the POJO generation, the type wiring, the custom scalars mapping...</LI>
252 	 * <LI>A list of:</LI>
253 	 * <UL>
254 	 * <LI>GraphQL type name: The full controller class for this type is ignored, and won't be generated</LI>
255 	 * <LI>GraphQL type's field name: The method in the controller of this type, for this field, is ignored, and won't
256 	 * be generated. The field must be written like this: <code>{type name}.{field name}</code></LI>
257 	 * </UL>
258 	 * </UL>
259 	 * <P>
260 	 * The accepted separators for the values are: comma, space, carriage return, end of line, space, tabulation. At
261 	 * least one separator must exist between two values in the list. Here is a sample:
262 	 * </P>
263 	 * 
264 	 * <PRE>
265 	 *          <ignoredSpringMappings>Type1, Type2.field1
266 	 *          	Type3
267 	 *          	Type4.field2
268 	 *          </ignoredSpringMappings>
269 	 * </PRE>
270 	 * <P>
271 	 * For field mapping, there must be no separator other than '.' between the type name and the field name. For
272 	 * instance, the following type declaration are invalid: 'type .field', 'type. field'
273 	 * </P>
274 	 * <P>
275 	 * To implement the ignored mappings, you'll have to follow the [spring-graphql
276 	 * documentation](https://docs.spring.io/spring-graphql/reference/controllers.html).
277 	 * </P>
278 	 */
279 	@Parameter(property = "com.graphql_java_generator.mavenplugin.ignoredSpringMappings", defaultValue = GenerateServerCodeConfiguration.DEFAULT_IGNORED_SPRING_MAPPINGS)
280 	public String ignoredSpringMappings;

283 	 * <P>
284 	 * The <I>javaTypeForIDType</I> is the java class that is used in the generated code for GraphQL fields that are of
285 	 * the GraphQL ID type. The default value is <I>java.util.UUID</I>. Valid values are: java.lang.String,
286 	 * java.lang.Long and java.util.UUID.
287 	 * </P>
288 	 * <P>
289 	 * This parameter is only valid for the server mode. When generating the client code, the ID is always generated as
290 	 * a String type, as recommended in the GraphQL doc.
291 	 * </P>
292 	 * <P>
293 	 * In other words: when in server mode and <I>javaTypeForIDType</I> is not set, all GraphQL ID fields are UUID
294 	 * attributes in java. When in server mode and <I>javaTypeForIDType</I> is set to the X type, all GraphQL ID fields
295 	 * are X attributes in java.
296 	 * </P>
297 	 * <P>
298 	 * Note: you can override this, by using the schema personalization capability. For more information, please have a
299 	 * look at the <A HREF=
300 	 * "https://github.com/graphql-java-generator/graphql-maven-plugin-project/wiki/usage_schema_personalization">Schema
301 	 * Personalization doc page</A>.
302 	 * </P>
303 	 * 
304 	 * @return
305 	 */
306 	@Parameter(property = "com.graphql_java_generator.mavenplugin.javaTypeForIDType", defaultValue = GraphQLConfiguration.DEFAULT_JAVA_TYPE_FOR_ID_TYPE)
307 	public String javaTypeForIDType;
