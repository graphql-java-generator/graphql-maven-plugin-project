20  /**
21   * <P>
22   * The <I>generatePojo</I> goal generates all the java objects that match the provided GraphQL schema. It allows to work
23   * in Java with graphQL, in a schema first approach.
24   * </P>
25   * This goal generates:
26   * <UL>
27   * <LI>One java interface for each GraphQL `union` and `interface`</LI>
28   * <LI>One java class for each GraphQL `type` and `input` type, including the query, mutation and subscription (if any).
29   * If the GraphQL type implements an interface, then its java class implements this same interface</LI>
30   * <LI>One java enum for each GraphQL enum</LI>
31   * </UL>
32   * 
33   * <P>
34   * Every class, interface and their attributes are marked with the annotation from the <A HREF=
35   * "https://graphql-maven-plugin-project.graphql-java-generator.com/graphql-java-runtime/apidocs/com/graphql_java_generator/annotation/package-summary.html">GraphQL
36   * annotation</A> package. This allows to retrieve the GraphQL information for every class, interface and attribute, at
37   * runtime.
38   * 
39   * <P>
40   * It can run in two modes (see the <A HREF=
41   * "https://graphql-maven-plugin-project.graphql-java-generator.com/graphql-maven-plugin/generatePojo-mojo.html#mode">mode
42   * plugin parameter</A> for more information):
43   * </P>
44   * <UL>
45   * <LI><B>server</B>: In the server mode, only the GraphQL annotation are added. You can add the JPA annotation, with
46   * the <I>generateJPAAnnotation</I> plugin parameter set to true.</LI>
47   * <LI><B>client</B>: The client mode is the default one. This mode generates the same POJO as in server mode, with the
48   * addition of the <A HREF="https://github.com/FasterXML/jackson">Jackson</A> annotations. These annotations allows to
49   * serialize and unserialize the GraphQL POJO to and from JSON. And the <I>CustomJacksonDeserializers</I> utility class
50   * is generated, that allows to deserialize custom scalars and arrays.</LI>
51   * </UL>
52   * <P>
53   * The generated code needs the relevant dependencies. The dependencies to add depends on the value of the
54   * <code>copyRuntimeSources</code> plugin parameter that you use.
55   * </P>
56   * <P>
57   * If <B>false</B> (default value since 2.0, recommended), you must add the runtime dependency, for the client or
58   * server, depending on the <code>mode</code> parameter you choosed. So the needed dependency would one of these two:
59   * </P>
60   * 
61   * <PRE>
62  		<dependency>
63  			<groupId>com.graphql-java-generator</groupId>
64  			<artifactId>graphql-java-client-runtime</artifactId>
65  			<version>${graphql-plugin.version}</version>
66  		</dependency>
67   * </PRE>
68   * 
69   * <PRE>
70  		<dependency>
71  			<groupId>com.graphql-java-generator</groupId>
72  			<artifactId>graphql-java-server-runtime</artifactId>
73  			<version>${graphql-plugin.version}</version>
74  		</dependency>
75   * </PRE>
76   * <P>
77   * If <B>true</B>, you must add the runtime dependency, for the client or server, depending on the <code>mode</code>
78   * parameter you choosed. So the needed dependency would one of these two:
79   * </P>
80   * 
81   * <PRE>
82  		<dependency>
83  			<groupId>com.graphql-java-generator</groupId>
84  			<artifactId>graphql-java-client-dependencies</artifactId>
85  			<type>pom</type>
86  			<version>${graphql-plugin.version}</version>
87  		</dependency>
88   * </PRE>
89   * 
90   * <PRE>
91  		<dependency>
92  			<groupId>com.graphql-java-generator</groupId>
93  			<artifactId>graphql-java-server-dependencies</artifactId>
94  			<type>pom</type>
95  			<version>${graphql-plugin.version}</version>
96  		</dependency>
97   * </PRE>
98   * 
99   * @author etienne-sf
100  */

116 					@Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateServerCode.*") })
