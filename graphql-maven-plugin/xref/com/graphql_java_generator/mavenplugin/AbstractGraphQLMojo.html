4   package com.graphql_java_generator.mavenplugin;
5   
6   import org.apache.maven.plugins.annotations.Parameter;
7   
8   import com.graphql_java_generator.plugin.conf.GraphQLConfiguration;
9   import com.graphql_java_generator.plugin.conf.PluginMode;
10  
11  /**
12   * This class is the class that contains all parameters that needed for the {@link GraphQLMojo}, that is all parameters
13   * that are common to all generate code Mojos. The parameters from the generateServerCode are inherited from
14   * {@link AbstractGenerateServerCodeMojo}, and the parameters that are specific to the generateClientCode must be
15   * manually added here, as a class may not inherit from two superclasses.<BR/>
16   * This avoids to redeclare each common parameter in each Mojo, including its comment. When a comment is updated, only
17   * one update is necessary, instead of updating it in each Mojo.
18   * 
19   * @author etienne-sf
20   */
21  public abstract class AbstractGraphQLMojo extends AbstractGenerateServerCodeMojo implements GraphQLConfiguration {
22  
23  	/**
24  	 * <P>
25  	 * <I>(since 1.7.1 version) Default value is _true_ for 1.x version, and _false_ for version 2.0 and after.</I>
26  	 * </P>
27  	 * <P>
28  	 * If this parameter is set to true, the plugin generates a XxxxResponse class for each query/mutation/subscription,
29  	 * and (if separateUtilityClasses is true) Xxxx classes in the util subpackage. This allows to keep compatibility
30  	 * with code Developed with the 1.x versions of the plugin.
31  	 * </P>
32  	 * <P>
33  	 * The recommended way to use the plugin is to directly use the Xxxx query/mutation/subscription executor classes,
34  	 * where Xxxx is the query/mutation/subscription name defined in the GraphQL schema. To do this, set this parameter
35  	 * to _false_, and use the plugin as described in the
36  	 * <a href="https://github.com/graphql-java-generator/graphql-maven-plugin-project/wiki/client_spring">wiki client
37  	 * page</a>.
38  	 * </P>
39  	 */
40  	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateDeprecatedRequestResponse", defaultValue = GraphQLConfiguration.DEFAULT_GENERATE_DEPRECATED_REQUEST_RESPONSE)
41  	boolean generateDeprecatedRequestResponse;
42  
43  	/**
44  	 * The generation mode: either <I>client</I> or <I>server</I>. Choose client to generate the code which can query a
45  	 * graphql server or server to generate a code for the server side.
46  	 */
47  	@Parameter(property = "com.graphql_java_generator.mavenplugin.mode", defaultValue = GraphQLConfiguration.DEFAULT_MODE)
48  	PluginMode mode;
49  
50  	@Override
51  	public PluginMode getMode() {
52  		return mode;
53  	}
54  
55  	@Override
56  	public boolean isGenerateDeprecatedRequestResponse() {
57  		return generateDeprecatedRequestResponse;
58  	}
59  
60  	protected AbstractGraphQLMojo(Class<?> springConfigurationClass) {
61  		super(springConfigurationClass);
62  	}
63  }
