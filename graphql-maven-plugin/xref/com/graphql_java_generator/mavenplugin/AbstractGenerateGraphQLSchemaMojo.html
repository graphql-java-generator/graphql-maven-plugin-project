4   package com.graphql_java_generator.mavenplugin;
5   
6   import java.io.File;
7   import java.io.IOException;
8   import java.util.Arrays;
9   import java.util.Map;
10  
11  import javax.inject.Inject;
12  
13  import org.apache.maven.plugins.annotations.Parameter;
14  import org.apache.maven.project.MavenProjectHelper;
15  
16  import com.graphql_java_generator.plugin.conf.GenerateGraphQLSchemaConfiguration;
17  
18  /**
19   * @author etienne-sf
20   */
21  public abstract class AbstractGenerateGraphQLSchemaMojo extends AbstractCommonMojo
22  		implements GenerateGraphQLSchemaConfiguration {
23  
24  	/** The encoding for the generated resource files */
25  	@Parameter(property = "com.graphql_java_generator.mavenplugin.resourceEncoding", defaultValue = GenerateGraphQLSchemaConfiguration.DEFAULT_RESOURCE_ENCODING)
26  	String resourceEncoding;
27  
28  	/** The folder where the generated GraphQL schema will be stored */
29  	@Parameter(property = "com.graphql_java_generator.mavenplugin.targetFolder", defaultValue = "${project.build.directory}"
30  			+ GenerateGraphQLSchemaConfiguration.DEFAULT_TARGET_FOLDER)
31  	File targetFolder;
32  
33  	/**
34  	 * The name of the target filename, in which the schema is generated. This file is stored in the folder, defined in
35  	 * the <I>targetFolder</I> plugin parameter.
36  	 */
37  	@Parameter(property = "com.graphql_java_generator.mavenplugin.targetSchemaFileName", defaultValue = GenerateGraphQLSchemaConfiguration.DEFAULT_TARGET_SCHEMA_FILE_NAME)
38  	String targetSchemaFileName;
39  
40  	@Inject
41  	protected MavenProjectHelper projectHelper;
42  
43  	AbstractGenerateGraphQLSchemaMojo(Class<?> springConfigurationClass) {
44  		super(springConfigurationClass);
45  	}
46  
47  	@Override
48  	public File getSchemaFileFolder() {
49  		return schemaFileFolder;
50  	}
51  
52  	@Override
53  	public String getSchemaFilePattern() {
54  		return schemaFilePattern;
55  	}
56  
57  	@Override
58  	public String getResourceEncoding() {
59  		return resourceEncoding;
60  	}
61  
62  	@Override
63  	public File getTargetFolder() {
64  		return targetFolder;
65  	}
66  
67  	@Override
68  	public String getTargetSchemaFileName() {
69  		return targetSchemaFileName;
70  	}
71  
72  	@Override
73  	public Map<String, String> getTemplates() {
74  		return templates;
75  	}
76  
77  	@Override
78  	public boolean isAddRelayConnections() {
79  		return addRelayConnections;
80  	}
81  
82  	@Override
83  	protected void executePostExecutionTask() throws IOException {
84  		String generatedResourceFolder = getTargetFolder().getPath();
85  		getLog().debug("Adding the generated resources folder: " + generatedResourceFolder);
86  		// Method 2 (should work better):
87  		projectHelper.addResource(project, generatedResourceFolder, Arrays.asList("**/*"), null);
88  		buildContext.refresh(getTargetFolder());
89  	}
90  }
