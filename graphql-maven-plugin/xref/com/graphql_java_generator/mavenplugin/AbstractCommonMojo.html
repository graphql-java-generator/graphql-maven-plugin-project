26   * This class is the super class of all Mojos. It contains all parameters that are common to all goals, and the
27   * {@link #execute()} method.<BR/>
28   * This avoids to redeclare each common parameter in each Mojo, including its comment. When a comment is updated, only
29   * one update is necessary, instead of updating it in each Mojo.
30   * 
31   * @author etienne-sf
32   */

362 	/** {@inheritDoc} */
363 	@Override
364 	final public void execute() throws MojoExecutionException, MojoFailureException {
365 
366 		try {
367 			LoggerFactory.getLogger(getClass()).debug("Starting generation of java classes from graphqls files");
368 
369 			// We'll use Spring IoC
370 			ctx = new AnnotationConfigApplicationContext();
371 			ctx.getBeanFactory().registerSingleton("mojo", this);
372 			ctx.register(springConfigurationClass);
373 			ctx.refresh();
374 
375 			// Let's log the current configuration (this will do something only when in
376 			// debug mode)
377 			ctx.getBean(CommonConfiguration.class).logConfiguration();
378 
379 			// Let's execute the job
380 			PluginExecutor executor = ctx.getBean(PluginExecutor.class);
381 			executor.execute();
382 
383 			executePostExecutionTask();
384 
385 			ctx.close();
386 
387 		} catch (Exception e) {
388 			throw new MojoExecutionException(e.getMessage(), e);
389 		}
390 	}
