20  /**
21   * <P>
22   * The <I>generateClientCode</I> Maven goal (and Gradle task) generates the java code from one or more GraphQL schemas.
23   * It allows to work in Java with graphQL, in a schema first approach.
24   * </P>
25   * It generates a class for each query, mutation and subscription type. These classes contain the methods to call the
26   * queries, mutations and subscriptions. That is: to execute a query against the GraphQL server, you just have to call
27   * one of these methods. It also generates the POJOs from the GraphQL schema. The <B>GraphQL response is stored in these
28   * POJOs</B>, for an easy and standard use in Java. <BR/>
29   * <BR/>
30   * You'll find more info in the tutorials: take a look at the
31   * <A HREF="https://github.com/graphql-java-generator/GraphQL-Forum-Maven-Tutorial-client">Maven client tutorial</A> or
32   * the <A HREF="https://github.com/graphql-java-generator/GraphQL-Forum-Gradle-Tutorial-client">Gradle client
33   * tutorial</A>
34   * 
35   * @author etienne-sf
36   */
37  @Mojo(name = "generateClientCode", defaultPhase = LifecyclePhase.GENERATE_SOURCES, requiresProject = true, threadSafe = true)
38  @ThreadSafe
39  // No need to add the @Component spring annotation: AbstractCommonMojo added this instance into the spring context, to
40  // use the instance which attributs has been set with the pom content

43  	// All the Mojo parameters are defined in the AbstractXxxx classes, that contains the contain the hierarchical
44  	// structure of the Maven goals.
45  	// See the explanation in the AbstractCommonMojo for more details.
46  
47  	@Configuration
48  	@Import({ JacksonAutoConfiguration.class })
49  	@ComponentScan(basePackageClasses = { DocumentParser.class, GraphqlUtils.class }, //
50  			excludeFilters = { @Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateGraphQLSchema.*"),
51  					@Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateServerCode.*"),
52  					@Filter(type = FilterType.REGEX, pattern = ".*\\.GraphQL.*") })

54  
55  	}
