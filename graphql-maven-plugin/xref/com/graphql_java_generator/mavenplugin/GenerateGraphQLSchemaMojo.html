18  /**
19   * <P>
20   * The <I>generateGraphQLSchema</I> goal generates GraphQL schema, based on the source GraphQL schemas, and possibly
21   * containing additional stuff, like the Relay connection objects.
22   * </P>
23   * It can be used to:
24   * <UL>
25   * <LI>Generate several GraphQL schema files into one file, for instance with additional schema files that would use the
26   * <I>extend</I> GraphQL keyword</LI>
27   * <LI>Reformat the schema file</LI>
28   * <LI>Generate the GraphQL schema with the Relay Connection stuff (Node interface, XxxEdge and XxxConnection types),
29   * thanks to the <I>addRelayConnections</I> plugin parameter.
30   * </UL>
31   * <P>
32   * This goal is, by default, attached to the Initialize maven phase, to be sure that the GraphQL schema are generated
33   * before the code generation would need it, if relevant.
34   * </P>
35   * 
36   * @author etienne-sf
37   */
38  @Mojo(name = "generateGraphQLSchema", defaultPhase = LifecyclePhase.INITIALIZE, requiresProject = true, threadSafe = true)
39  @ThreadSafe
40  // No need to add the @Component spring annotation: AbstractCommonMojo added this instance into the spring context, to
41  // use the instance which attributs has been set with the pom content
42  public class GenerateGraphQLSchemaMojo extends AbstractGenerateGraphQLSchemaMojo {
43  
44  	// All the Mojo parameters are defined in the AbstractXxxx classes, that contains the contain the hierarchical
45  	// structure of the Maven goals.
46  	// See the explanation in the AbstractCommonMojo for more details.
47  
48  	@Configuration
49  	@ComponentScan(basePackageClasses = { DocumentParser.class, GraphqlUtils.class }, //
50  			excludeFilters = { @Filter(type = FilterType.REGEX, pattern = ".*\\.GraphQL.*"),
51  					@Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateClientCode.*"),
52  					@Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateCode.*"),
53  					@Filter(type = FilterType.REGEX, pattern = ".*\\.GenerateServerCode.*") })
54  	public class SpringConfiguration {
55  
56  	}
57  
58  	GenerateGraphQLSchemaMojo() {
59  		super(SpringConfiguration.class);
60  	}
61  
62  }
