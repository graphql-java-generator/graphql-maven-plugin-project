1   package com.graphql_java_generator.mavenplugin;
2   
3   import org.apache.maven.plugins.annotations.Parameter;
4   
5   import com.graphql_java_generator.plugin.conf.GeneratePojoConfiguration;
6   import com.graphql_java_generator.plugin.conf.PluginMode;
7   
8   public abstract class AbstractGeneratePojoMojo extends AbstractGraphQLMojo implements GeneratePojoConfiguration {
9   
10  	/**
11  	 * <P>
12  	 * The <A HREF="https://github.com/FasterXML/jackson">Jackson</A> annotations are necessary to properly deserialize
13  	 * the json, that is incoming from the GraphQL Server. Setting this property to false allows to not generate them.
14  	 * </P>
15  	 * <P>
16  	 * If this property is set to true, the Jackson annotations are added in the generated GraphQL objects. The
17  	 * <A HREF="https://github.com/FasterXML/jackson">Jackson</A> dependencies must then be added to the target project,
18  	 * so that the project compiles.
19  	 * </P>
20  	 * <P>
21  	 * The default value is:
22  	 * </P>
23  	 * <UL>
24  	 * <LI><I>true</I> when in <I>client</I> mode.</LI>
25  	 * <LI><I>false</I> when in <I>server</I> mode.</LI>
26  	 * </UL>
27  	 */
28  	@Parameter(property = "com.graphql_java_generator.mavenplugin.generateJacksonAnnotations", defaultValue = GeneratePojoConfiguration.DEFAULT_GENERATE_JACKSON_ANNOTATIONS)
29  	public Boolean generateJacksonAnnotations;
30  
31  	@Override
32  	public boolean isGenerateJacksonAnnotations() {
33  		if (generateJacksonAnnotations != null) {
34  			return generateJacksonAnnotations;
35  		} else
36  			return PluginMode.client.equals(getMode());
37  	}
38  
39  	protected AbstractGeneratePojoMojo(Class<?> springConfigurationClass) {
40  		super(springConfigurationClass);
41  	}
42  
43  }
